    \item[\textcolor{parryblue}{AI Validator}] Post-detection validation to reduce false positives (55\% â†’ 25\%)

\section{Implementation Guide}
% ============================================================

\subsection{Quick Start (5 Minutes)}

\begin{verbatim}
# 1. Install Ollama (LLM runtime)
brew install ollama
brew services start ollama

# 2. Download AI model (one-time, ~4GB)
ollama pull codellama:7b-instruct-q4_K_M

# 3. Install Parry
pip install parry-scanner

# 4. Verify installation
parry doctor

# 5. Run first scan
parry scan /path/to/code --mode deep --validate
\end{verbatim}
